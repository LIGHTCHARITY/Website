        // Charity Adding Function

        function addCharity() {

            const charityName = document.getElementById('charityName').value.trim();

            if (!charityName) {

                alert("Please enter a charity name!");

                return;

            }

            const listItem = document.createElement('li');

            listItem.textContent = charityName;

            document.getElementById('charityList').appendChild(listItem);

            document.getElementById('charityName').value = '';

        }


        // Image Preview Function

        function previewImage(event) {

            const reader = new FileReader();

            reader.onload = function () {

                const img = document.createElement('img');

                img.src = reader.result;

                document.getElementById('imagePreview').innerHTML = '';

                document.getElementById('imagePreview').appendChild(img);

            }

            reader.readAsDataURL(event.target.files[0]);

        }


        // Form Submission Handler

        function submitForm(event) {

            event.preventDefault();

            alert('Form submitted successfully!');

            document.getElementById('contactForm').reset();

        }


        // Content Section Visibility Animation

        window.addEventListener('scroll', function () {

            const sections = document.querySelectorAll('.content-section');

            sections.forEach(function (section) {

                if (isElementInView(section)) {

                    section.classList.add('visible');

                }

            });

        });


        function isElementInView(element) {

            const rect = element.getBoundingClientRect();

            return rect.top <= window.innerHeight && rect.bottom >= 0;

        }
